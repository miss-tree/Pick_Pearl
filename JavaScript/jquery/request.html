JQ_Ajax

	$.ajax({
		type:"GET",
		url:baseUrl+"/getShopList",
		success:function(res){代表从服务器接收回来的数据
			data=res.result;
			放在回调函数执行,才能够拿到正确数据
			console.log(data);
		},
		JQ可以直接写入请求头(方法一)
		headers:{
			"authority":"bluej"
		},
		请求拦截器
		beforeSend:function(xhr){
			console.log("beforeSend");
			console.log(xhr);
			将token写入请求头（方法二）
			xhr.setRequestHeader("authority","bluej");
		},
		设置请求超时时间,保证用户体验
		timeout:1000,
		error:function(){
			console.log("请求失败")
		}
	});
