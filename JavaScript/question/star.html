实现空心菱形

	    *

	    *			
	

	for(var i=0;i <=8;i++){
		for(var j=1;j <=9;j++){
			if(i <=4){
				if((j==5+i||j==5-i)){//凡是中轴+-当前行号
					document.write("*");
				}else{
					document.write(" <i >@ </i >");
				}
			}else{
				if((j==5+(8-i)||j==5-(8-i))){//凡是中轴+-当前行号
					document.write("*");
				}else{
					document.write(" <i >@ </i >");
				}
			}
		}
		document.write(" <br/ >");
	}
