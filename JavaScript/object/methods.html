defineProperty()

		let obj = {
		  name: '子君',
		  officialAccounts: '前端有的玩'
		}
		Object.defineProperty(obj,'name', {
		  // 设置writable 是 false, 这个属性将不能被修改
		  writable: false
		})
		// 修改obj.name
		obj.name = "君子"
