第一

	document.write("<table border=1>");
	for(var j=1;j<=9;j++){
		document.write("<tr>");
		for(var i=1;i<=9;i++){
			if(i>j){
				document.write("<td></td>")
			}else{
				document.write("<td>"+i+"*"+j+"="+i*j+"</td>");
			}
		};
		document.write("</tr>");
	}
	document.write("</table>");

第二

	document.write('<table border="1">');
	for(var a = 9; a > 0; a--) {
		document.write('<tr>');
		for(var b = 1; b <= 9; b++) {
			document.write('<td >');
			if(b <= a) {
				document.write(b + '*' + a + '=' + a * b);
			} else {
				document.write('');
			}
			document.write('</td>');
		}
		document.write('</tr>');
	}
	document.write('</table>');
