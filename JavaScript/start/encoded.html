escape(编码) 和 unescape(解码)

	var url = "http://www.baidu.com/s?";	
	url = url + escape("name=小米");
	alert(url); =>"http://www.baidu.com/s?name%3D%u5C0F%u7C73"
	alert(unescape(url));=>"http://www.baidu.com/s?name=小米"

encodeURIComponent(url) 函数 decodeURIComponent(url) 函数

	var url = "http://www.baidu.com/s?";	
	url = url + encodeURIComponent("name=小米");
	alert(url); =>"http://www.baidu.com/s?name%3D%E5%B0%8F%E7%B1%B3"
	alert(decodeURIComponent(url));=>"http://www.baidu.com/s?name=小米"
