Dayjs

Day.js 是一个轻量的处理时间和日期的 JavaScript 库， 和 Moment.js 的 API 设计保持完全一样。 中文文档地址

可以设置取值，赋值、验证、复制、查询 

var now=new Date();
console.log(now.getMonth()+1)==>获取当前月份
var month=dayjs().month(0)==>设置月份
console.log(month)
tt=dayjs().month(now.getMonth()+3)
y=tt.year().toString()
m=tt.month().toString()
d=tt.date().toString()
console.log(y+"-"+m+"-"+d)
var last=y+"-"+m+"-"+d
var fultrue=(new Date(last)).getTime()-24*60*60*1000;
console.log(fultrue)
fultrue=new Date(fultrue)
//设置两个月后前一天
fultrue=`${fultrue.getFullYear()}-${fultrue.getMonth() + 1}-${fultrue.getDate()}`
console.log(fultrue)
