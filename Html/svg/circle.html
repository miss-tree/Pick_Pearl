		js		
		var sum=0;
		var timer=null;
		var num=314;
		var circle=document.querySelector("circle");
		timer=setInterval(function(){
			sum+=1;
			txt.innerText=sum+"%";
			circle.style.strokeDashoffset=num*3.14+"px";
			num--;
			if(sum==100){
				clearInterval(timer);
			}
		},100)

circle属性

  • cx,cy：定义圆的坐标位置
  • r：圆的半径
  • fill：填充色
  • stroke：描边色
  • stroke-width：边框宽度
