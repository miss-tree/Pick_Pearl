算术运算符( + - * / %)

	加、减、乘、除、求余，其中%是求余运算	
	var total = (1 + 4) * 5;	
	var i = 100;	
	var temp = (i – 20) / 2;		
	alert(“10”+ 20)		//return 1020;	
	alert(10 + 20)		//return 30;

后增量/后减量运算符 ++ ,--

	var i = 10;	
	var a = i++;		
	alert(a);	=>10

比较运算符 ( >, <, >=, <=, ==, !=，===,!== )

	alert ( 10 > 5 );	//outputs	 true		
	var i = 100;	var n = 100;	
	alert(i == n);	//outputs true;	
	alert(i != n);	//outputs false;          
	alert(i === n)         //数据类型，值相等

逻辑运算符( &&, || , !)

	&& ： 逻辑与	   	一个为真,则为true
	|| : 逻辑或		全部为真,才为true
	！ ：逻辑非		对现有结果进行取反
	var i = 8;	
	alert ( i<5 && i<10);		//outputs false	
	alert ( i > 100 || i < 10);   	//outputs true	
	alert(!(10 > 5));		//outputs false
