事件冒泡

DOM2级事件的"三个阶段"
  • (w3c) 捕获阶段-->处于目标阶段-->冒泡阶段
  • (IE) 处于目标阶段-->冒泡阶段

addEventListener()第三个参数表示触发事件的阶段,false(默认值:冒泡阶段触发),true(捕获阶段触发)

	//阻止事件冒泡
	e.stopPropagation()
	btn.addEventListener("click",function(e){
		alert("hello");
	},false);	
	
	e.preventDefault();		
	var btn = document.querySelector("a")	
	btn.addEventListener("click",function(e){
		e.preventDefault();	//阻止默认事件
		alert("world");
	},true);
