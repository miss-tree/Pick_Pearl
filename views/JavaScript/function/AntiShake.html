防抖函数

	const debounce = (cb, delay = 1000) => {
	  let timer = null;
	  return function (...args) {
		const context = this;
		if (timer) clearTimeout(timer);
		timer = setTimeout(() => {
		  cb.apply(context, args);
		  timer = null;
		}, delay);
	  }
	}

节流函数(使用时间戳)

	const throttleUseTimeStamp = (cb, delay = 1000) => {
	  let startTime = Date.now();  
	　return function(...args) {      
	　　const context = this;                        
	　　const now = Date.now();           
	　　if (now - startTime >= delay) {
	      cb.apply(context, args);
	      startTime = Date.now();        
	　　}
	　} 
	}

节流函数(使用定时器)

	const throttleUseTimer = (cb, delay) => {
	  let timer = null;
	  return function(...args) {
		const context = this;
		if (!timer) {
		  timer = setTimeout(() => {
			cb.apply(context, args);      
			timer = null;      
		  }, delay);  
		} 
	  }
	}
