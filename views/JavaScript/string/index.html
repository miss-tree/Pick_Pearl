charAt()

	var str="bbqLove";
	console.log(str.charAt(3));=>"q"

substr()

	var str="bbqLove";
	console.log(str.charAt(2,3));=>"qLo"	
	
	第一个参数  截取的位置
	第二个参数  截取的长度  (没有指定的长度会默认全部)

substring()

	var str="bbqLove";
	console.log(str.charAt(0,3));=>"bbq"
					
	第一个参数  截取的开始位置
	第二个参数  截取的结束位置

indexOf()、lastIndexOf()

indexOf()：找到第一个查询字符串的下标
lastIndexOf()：找到最后查询字符串的下标
	var str="Hello world, welcome to the universe.";
	console.log(str.indexOf("o"));=>4
	alert(str.lastIndexOf("o"));=>22

replace()

	var str1="abc123";
	str1=str1.replace("b","B");
	console.log(str1);=>"aBc123"
	
	去除空格
	str=str.replace(/\s*/g,'') 	=>去除所有
	str=str.replace(/^\s*/g,'') 	=>去除左边
	str=str.replace(/(\s*$)/g,'') 	=>去除右边
	str=str.trim() 	=>去除两边

concat()

	var str1="abc123";
	var str2="def456";
	var str3=str1.concat(str2);
	console.log(str3);=>"abc123def456"

search()

	var str="Visit W3CSchool!";
	var n=str.search("W3CSchool")	//没有则为-1，和indexof在此类似
	console.log(n)=>6

split()

	var str="Visit W3CSchool!";
	var newArr=str.split('')	
	=> ["V", "i", "s", "i", "t", " ", "W", "3", "C", "S", "c", "h", "o", "o", "l", "!"]
	newString=newArr.join('')
	=>"Visit W3CSchool!"
