系统消息框 alert()

alert('hello');    //优先执行，执行完后再执行其他代码

确认对话框 confirm()

	该方法返回一个boolean值，如果点击ok返回true,点击cancel返false;
	if(confirm("确定要删除吗?")){	
		//删除	
	}

输入对话框 prompt()

	如果点击ok将文本框中的值作为函数值返回，如果点击cancel返回null
	var name = prompt("请输入你的姓名？","");
	if(name != null){	
		alert(‘welcome’+ name);
	}

打开新窗口 window.open()

	window.open("http://www.baidu.com","_blank","width=300, height=200")	
	只有在打开新窗口的情况下后面设置的尺寸才有效，其他 "_self"、"parent"、"top"、"_search"都无效
